  // Only change code below this line
  if(card == 2 || card == 3 || card == 4 || card == 5 || card ==6)
  count++;
  else if(card == 10 || card == 'J' || card == 'Q' || card == 'K' || card == 'A')
  count--;
  let x = "";
  if(count > 0)
  x = "Bet";
  else 
  x = "Hold";

}
